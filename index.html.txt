        function construirPayload(chave, nome, cidade, valor, txid) {
            const valorStr = valor.toFixed(2);
            const mai = '0014BR.GOV.BCB.PIX01' + String(chave.length).padStart(2, '0') + chave;
            const maiFull = '26' + String(mai.length).padStart(2, '0') + mai;
            
            let payload = '000201' + maiFull + '52040000530398654' + String(valorStr.length).padStart(2, '0') + valorStr;
            payload += '5802BR59' + String(nome.length).padStart(2, '0') + nome;
            payload += '60' + String(cidade.length).padStart(2, '0') + cidade;
            payload += '62' + String(4 + txid.length).padStart(2, '0') + '05' + String(txid.length).padStart(2, '0') + txid;
            payload += '6304';
            
            // CRC16-CCITT-FALSE
            let crc = 0xFFFF;
            for (let i = 0; i < payload.length; i++) {
                crc ^= payload.charCodeAt(i) << 8;
                for (let j = 0; j < 8; j++) crc = (crc & 0x8000) ? ((crc << 1) ^ 0x1021) : (crc << 1);
                crc &= 0xFFFF;
            }
            return payload + crc.toString(16).toUpperCase().padStart(4, '0');
        }
